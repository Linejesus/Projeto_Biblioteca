CREATE TABLE usuarios (
    id_usuario SERIAL PRIMARY KEY NOT NULL UNIQUE,
    nome VARCHAR(150) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    senha VARCHAR(255) NOT NULL,
    cpf VARCHAR(14) NOT NULL UNIQUE,
    data_cadastro DATE DEFAULT CURRENT_DATE
);


CREATE TABLE bibliotecas (
    id_biblioteca SERIAL PRIMARY KEY NOT NULL UNIQUE,
    nome VARCHAR(150) NOT NULL,
    estado VARCHAR(100) NOT NULL,
    cidade VARCHAR(100) NOT NULL,
    bairro VARCHAR(100) NOT NULL,
    rua VARCHAR(100) NOT NULL,
    numero INT NOT NULL
);


CREATE TABLE emprestimos (
    id_emprestimo SERIAL PRIMARY KEY NOT NULL UNIQUE,
    id_usuario INT NOT NULL,
    id_livro INT NOT NULL,
    id_biblioteca INT NOT NULL,

    data_emprestimo DATE DEFAULT CURRENT_DATE,
    data_devolucao_prevista DATE NOT NULL DEFAULT (CURRENT_DATE + INTERVAL '15 days'),
    data_devolucao_real DATE,

    status VARCHAR(20) DEFAULT 'emprestado',
    renovacoes INT DEFAULT 0,

    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario),
    FOREIGN KEY (id_biblioteca) REFERENCES bibliotecas(id_biblioteca)
);


CREATE TABLE multas (
    id_multa SERIAL PRIMARY KEY NOT NULL UNIQUE,
    id_emprestimo INT NOT NULL,
    valor_total DECIMAL(8,2) NOT NULL,
    data_multa DATE NOT NULL DEFAULT CURRENT_DATE,
    dias_atraso INT,
    paga BOOLEAN DEFAULT FALSE,
    data_pagamento DATE,

    FOREIGN KEY (id_emprestimo) REFERENCES emprestimos(id_emprestimo)
);
